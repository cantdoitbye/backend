[Unit]
Description=gunicorn daemon
Requires=gunicorn.socket
After=network.target

[Service]
User=root
Group=www-data
WorkingDirectory=/root/ooumph_social_backend_neo4j
ExecStart=/root/ooumph_social_backend_neo4j/oenv/bin/gunicorn \
          --access-logfile - \
          --workers 3 \
          --bind unix:/run/gunicorn.sock \
          socialooumph.wsgi:application

[Install]
WantedBy=multi-user.target
